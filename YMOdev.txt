def yazdir(text,adet):

# 2 dikdortgenin alan ve cevresini hesaplayan fonk yaziniz

def dikdortgen(k1,k2):
    c=2*(k1+k2)
    a=k1*k2
    return a,c

print(dikdortgen(4,5))

# 3 yazi tura uygulamasini fonk kullanarak yapiniz.
def yaziTura():
    import random
    sayi = random.random()

    if sayi > 0.5:
        return "Tura"
    else:
        return "Yazi"
    
sonuc = yaziTura()
